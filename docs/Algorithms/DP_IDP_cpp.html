 17    """
 18    Class to implement the DP_IDP algorithm for ranking and sorting data based on multiple criteria.

 20    def __init__(self, r):
 21        """
 22        Initialize the DP_IDP algorithm with the given relation.

 28    def run(self):
 29        """
 30        Run the DP_IDP algorithm to compute the ranking and sorting of data based on multiple criteria.
 31        """
 32        self.dom, self.sp = self.calculMatriceDesDominants(self.r)
 33        self.dom, self.sky, self.spTot, self.skyCard, self.domCard = self.calculGrapheDeCouverture(self.dom, self.sp)
 34        self.sky = self.calculLm(self.dom, self.sky, self.skyCard)

 37    def calculMatriceDesDominants(self, r):
 38        """
 39        Calculate the dominance matrix and the skyline points.
 40        :param r: the relation to be processed
 41        :return: dominance matrix and skyline points
 42        """
 43        len_ = len(r)
 44        dom = [[False for y in range(len_)] for x in range(len_)]
 45        sp = [True] * len_
 46        r_keys = list(r.keys())
 47        col_len = len(r.get(r_keys[0]))
 48        for row_index in range(len_):
 49            for col_index in range(len_):
 50                if row_index != col_index:
 51                    sup = True
 52                    for k in range(col_len):
 53                        r_val = list(r.values())
 54                        if r_val[col_index][k] > r_val[row_index][k]:
 55                            sup = False
 56                            break
 57                    if sup:
 58                        dom[row_index][col_index] = True
 59                        sp[row_index] = False
 60                    else:
 61                        dom[row_index][col_index] = False
 62
 63        return dom, sp

 65    def calculGrapheDeCouverture(self, dom, sp, display_graph=False):
 66        """
 67        Calculate the coverage graph and the skyline points.
 68        :param dom: dominance matrix
 69        :param sp: skyline points
 70        :param display_graph: display the graph if True
 71        :return: dominance matrix, coverage graph, total skyline points, skyline cardinality, and dominance cardinality
 72        """
 73        n = len(dom)
 74        sky = {}
 75        skyCard = {}
 76        domCard = [0] * n
 77        spTot = 0
 78        for i in range(n):
 79            if sp[i]:
 80                sky[i] = [0] * n
 81                spTot += 1
 82                skyCard[i] = 0
 83            else:
 84                for j in range(n):
 85                    if i != j and dom[i][j]:
 86                        if j in sky.keys():
 87                            sky[j][i] = 1
 88                            skyCard[j] += 1
 89                            domCard[i] += 1
 90                        if not sp[j]:
 91                            for k in range(n):
 92                                if dom[j][k]:
 93                                    dom[i][k] = False
 94        return dom, sky, spTot, skyCard, domCard

 96    def calculLm(self, dom, sky, skyCard):
 97        n = len(dom)
 98        dom_flat = [item for sublist in dom for item in sublist]
 99        sky_flat = [item for sublist in sky.values() for item in sublist]
100        skyCard_list = [skyCard[i] for i in sky.keys()]
101
102        dom_ffi = ffi.new("bool[]", dom_flat)
103        sky_ffi = ffi.new("int[]", sky_flat)
104        skyCard_ffi = ffi.new("int[]", skyCard_list)
105
106        C.calculLm(dom_ffi, sky_ffi, skyCard_ffi, n)
107
108        # Récupérer les résultats
109        updated_sky = np.array(ffi.unpack(sky_ffi, n * n)).reshape(n, n)
110        sky_result = {i: list(updated_sky[i]) for i in sky.keys()}
111        return sky_result

113    def calculScore(self, sky, domCard, spTot):
114        import math
115        score = {}
116        for i in sky:
117            tot = 0
118            for j in range(len(sky[i])):
119                if sky[i][j] > 0:
120                    tot += 1 / sky[i][j] * math.log10(spTot / domCard[j])
121            score[i] = tot
122        return score

 38    def calculMatriceDesDominants(self, r):
 39        """
 40        Calculate the dominance matrix and the skyline points.
 41        :param r: the relation to be processed
 42        :return: dominance matrix and skyline points
 43        """
 44        len_ = len(r)
 45        dom = [[False for y in range(len_)] for x in range(len_)]
 46        sp = [True] * len_
 47        r_keys = list(r.keys())
 48        col_len = len(r.get(r_keys[0]))
 49        for row_index in range(len_):
 50            for col_index in range(len_):
 51                if row_index != col_index:
 52                    sup = True
 53                    for k in range(col_len):
 54                        r_val = list(r.values())
 55                        if r_val[col_index][k] > r_val[row_index][k]:
 56                            sup = False
 57                            break
 58                    if sup:
 59                        dom[row_index][col_index] = True
 60                        sp[row_index] = False
 61                    else:
 62                        dom[row_index][col_index] = False
 63
 64        return dom, sp

 66    def calculGrapheDeCouverture(self, dom, sp, display_graph=False):
 67        """
 68        Calculate the coverage graph and the skyline points.
 69        :param dom: dominance matrix
 70        :param sp: skyline points
 71        :param display_graph: display the graph if True
 72        :return: dominance matrix, coverage graph, total skyline points, skyline cardinality, and dominance cardinality
 73        """
 74        n = len(dom)
 75        sky = {}
 76        skyCard = {}
 77        domCard = [0] * n
 78        spTot = 0
 79        for i in range(n):
 80            if sp[i]:
 81                sky[i] = [0] * n
 82                spTot += 1
 83                skyCard[i] = 0
 84            else:
 85                for j in range(n):
 86                    if i != j and dom[i][j]:
 87                        if j in sky.keys():
 88                            sky[j][i] = 1
 89                            skyCard[j] += 1
 90                            domCard[i] += 1
 91                        if not sp[j]:
 92                            for k in range(n):
 93                                if dom[j][k]:
 94                                    dom[i][k] = False
 95        return dom, sky, spTot, skyCard, domCard

 97    def calculLm(self, dom, sky, skyCard):
 98        n = len(dom)
 99        dom_flat = [item for sublist in dom for item in sublist]
100        sky_flat = [item for sublist in sky.values() for item in sublist]
101        skyCard_list = [skyCard[i] for i in sky.keys()]
102
103        dom_ffi = ffi.new("bool[]", dom_flat)
104        sky_ffi = ffi.new("int[]", sky_flat)
105        skyCard_ffi = ffi.new("int[]", skyCard_list)
106
107        C.calculLm(dom_ffi, sky_ffi, skyCard_ffi, n)
108
109        # Récupérer les résultats
110        updated_sky = np.array(ffi.unpack(sky_ffi, n * n)).reshape(n, n)
111        sky_result = {i: list(updated_sky[i]) for i in sky.keys()}
112        return sky_result

114    def calculScore(self, sky, domCard, spTot):
115        import math
116        score = {}
117        for i in sky:
118            tot = 0
119            for j in range(len(sky[i])):
120                if sky[i][j] > 0:
121                    tot += 1 / sky[i][j] * math.log10(spTot / domCard[j])
122            score[i] = tot
123        return score
