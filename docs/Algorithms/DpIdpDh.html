322    print()
323    col_len = len(r.get(r_keys[0]))
324    for row_index in range(len_):
325        for col_index in range(len_):
326            if row_index != col_index:
327                sup = True
328                for k in range(col_len):
329                    r_val = list(r.values())
330                    if r_val[col_index][k] > r_val[row_index][k]:
331                        sup=False
332                        break
333                if sup:
334                    dom[row_index][col_index] = True
335                    sp[row_index] = False
336                else:

696        """
697        Function to calculate the lm of the skyline points using a stack instead of recursion.
698        :param dom: the dominance matrix
699        :param sky: the skyline points

701        :param i: the index of the skyline point

724    def lm(self, dom, sky, skyCard, i, j, prof):
725        """
726        Recursive function to calculate and update the dominance levels (lm)
727        of skyline points.
728
729        When a point j is dominated by another point k, this function updates
730        the 'sky' structure and decrements the corresponding cardinality.
731        The update is propagated recursively with an incremented dominance level.
732
733        :param dom: list[list[bool]] - dominance matrix (dom[k][j] == True means k dominates j)
734        :param sky: dict - structure representing the skyline points
735        :param skyCard: dict - dictionary containing the remaining dominance count for each point
736        :param i: int - index of the current point being processed in sky
737        :param j: int - target index in sky
738        :param prof: int - current dominance level
739        :yield: dict - updated version of the sky structure
740        """
741        n = len(dom)
742
743        for k in range(n):
744            if dom[k][j]:
745                if sky[i][k] > 0 and skyCard[i] > 0:
746                    dom[k][j] = False
747                    value = sky[i][k]
748
749                    # Prioritize the most probable condition first
750                    if value == 1 or value > prof:
751                        sky[i][k] = prof
752
753                    skyCard[i] -= 1
754
755                    if skyCard[i] == 0:
756                        yield sky
757                        return  # Stop recursion once all dominances are handled
758
759                    new_prof = prof + 1
760
761                    # Recursive call and iteration over updated skyline states
762                    for updated_sky in self.lm(dom, sky, skyCard, i, k, new_prof):
763                        yield updated_sky
764
765        yield sky

769        """

801        """
802        Calculates a score for each skyline entry based on dominance levels and frequency.
803
804        The score is computed as a sum over all dimensions where the skyline value > 0:
805        (1 / level) * log10(total support / cardinality of dominators)
806
807        :param sky: dict - skyline structure (matrix-like: sky[i][j] gives dominance level of j for skyline i)

815            for j in range(len(sky[i])):
816                if sky[i][j] > 0:
817                    tot += 1 / sky[i][j] * math.log10(spTot / domCard[j])
818            score[i + 1] = tot
819        return score

726        """
727        Recursive function to calculate and update the dominance levels (lm)
728        of skyline points.
729
730        When a point j is dominated by another point k, this function updates
731        the 'sky' structure and decrements the corresponding cardinality.
732        The update is propagated recursively with an incremented dominance level.
733
734        :param dom: list[list[bool]] - dominance matrix (dom[k][j] == True means k dominates j)
735        :param sky: dict - structure representing the skyline points
736        :param skyCard: dict - dictionary containing the remaining dominance count for each point
737        :param i: int - index of the current point being processed in sky
738        :param j: int - target index in sky
739        :param prof: int - current dominance level
740        :yield: dict - updated version of the sky structure
741        """
742        n = len(dom)
743
744        for k in range(n):
745            if dom[k][j]:
746                if sky[i][k] > 0 and skyCard[i] > 0:
747                    dom[k][j] = False
748                    value = sky[i][k]
749
750                    # Prioritize the most probable condition first
751                    if value == 1 or value > prof:
752                        sky[i][k] = prof
753
754                    skyCard[i] -= 1
755
756                    if skyCard[i] == 0:
757                        yield sky
758                        return  # Stop recursion once all dominances are handled
759
760                    new_prof = prof + 1
761
762                    # Recursive call and iteration over updated skyline states
763                    for updated_sky in self.lm(dom, sky, skyCard, i, k, new_prof):
764                        yield updated_sky
765
766        yield sky
