181    def run_algorithm(self):
182        try:
183            dtype = self.dataCombo.currentText()
184            algo_name = self.algoCombo.currentText()
185            out_fmt = self.outputCombo.currentText()
186
187            data = self.load_data(dtype)
188            nb_cols = self.count_columns(data, dtype)
189
190            prefs = None
191            if algo_name in ("RankSky", "CoskyAlgorithme", "CoskySQL"):

193                if not ok or not s: return
194                parts = [p.strip().lower() for p in s.split(",")]
195                if len(parts) != nb_cols or any(p not in ("min", "max") for p in parts):
196                    QMessageBox.critical(self, "Error", "Incorrect preference list.")
197                    return
198                prefs = [Preference.MIN if p == "min" else Preference.MAX for p in parts]
199
200            algo_cls = {"SkyIR": SkyIR, "DpIdpDh": DpIdpDh,
201                        "CoskyAlgorithme": CoskyAlgorithme, "CoskySQL": CoskySQL,
202                        "RankSky": RankSky}[algo_name]
203
204            exporter = CsvExporterImpl("../Assets/Export/CSVFiles/Results.csv") if out_fmt == "CSV" \
205                      else JsonExporterImpl("../Assets/Export/JsonFiles/Result.json")
206
207            app = App(data, algo_cls, exporter=exporter,
208                      input_type=dtype, input_file=self.selectedDataPath or "inline",
209                      preferences=prefs)
210            self.lastAppInstance = app
211            self.statusLabel.setText(f"Done in {app.execution_time}s")
212
213            all_data = self._extract_all_points(data)

218            QMessageBox.critical(self, "Error", str(e))

233    def get_skyline_points(self):
234        algo = self.lastAppInstance.algo
235        inst = self.lastAppInstance.algo_instance
236        result = []
237        if algo == "SkyIR": result = inst.result
238        elif algo == "DpIdpDh": result = inst.score

245    def display_graph(self, all_points, skyline_points):
246        self.figure.clear()
247        n_dim = len(all_points[0]) if all_points else 0
248        all_array = np.array(all_points)
249        sky_indices = []
250        for i, pt in enumerate(all_points):
251            for sp in skyline_points:
252                if np.allclose(pt[:3], sp[:3], atol=1e-6):
253                    sky_indices.append(i)
254                    break
255
256        if n_dim == 3:
257            ax = self.figure.add_subplot(111, projection='3d')
258            max_vals = np.max(all_array, axis=0)
259            norm_all = all_array / max_vals
260            norm_sky = norm_all[sky_indices]
261            ax.scatter(norm_all[:, 0], norm_all[:, 1], norm_all[:, 2], color='lightgray')
262            ax.scatter(norm_sky[:, 0], norm_sky[:, 1], norm_sky[:, 2], color='blue')
263            try:
264                hull = ConvexHull(norm_sky)
265                for s in hull.simplices:
266                    tri = norm_sky[s]
267                    ax.plot(tri[:, 0], tri[:, 1], tri[:, 2], color='red', linestyle='--')
268            except Exception as e:
269                print("Erreur dans le tracé de la surface Pareto:", e)
270            ax.set_xlabel("Dim 1")
271            ax.set_ylabel("Dim 2")
272            ax.set_zlabel("Dim 3")
273            ax.set_title("Skyline 3D (normalisé par max dimensionnel)")
274        else:
275            ax = self.figure.add_subplot(111)
276            reduced = PCA(n_components=2).fit_transform(StandardScaler().fit_transform(all_array))
277            all_x, all_y = reduced[:, 0], reduced[:, 1]
278            sky_x, sky_y = reduced[sky_indices, 0], reduced[sky_indices, 1]
279            ax.scatter(all_x, all_y, color='lightgray')
280            ax.scatter(sky_x, sky_y, color='blue')
281            skyline_proj = np.column_stack((sky_x, sky_y))
282            skyline_sorted = skyline_proj[np.argsort(skyline_proj[:, 0])]
283            ax.plot(skyline_sorted[:, 0], skyline_sorted[:, 1], color='red', linestyle='--')
284            ax.set_title("Skyline (bleu) vs Tous les points")
285            ax.set_xlabel("Composante 1")
286            ax.set_ylabel("Composante 2")
287        self.canvas.draw()
