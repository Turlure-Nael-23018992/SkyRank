198    def count_columns(self, obj, dtype):
199        if isinstance(obj, DictObject): return len(next(iter(obj.r.values())))
200        if isinstance(obj, JsonObject): return len(next(iter(readJson(obj.fp).values())))
201        if isinstance(obj, DbObject):
202            con = sqlite3.connect(obj.fp); cur = con.cursor()
203            cur.execute(f"SELECT * FROM {self.tableName}"); n = len(cur.fetchall()[0]) - 1
204            con.close(); return n
205        if dtype == "Generate Random Database": return int(self.columnEntry.text())
206        raise ValueError("Cannot count columns")

208    def run_algorithm(self):
209        """
210        Loads the selected dataset, collects user preferences if required,
211        instantiates the chosen algorithm and exporter, runs the App pipeline,
212        and triggers a graph update. Displays the execution time on success
213        or an error message on failure.
214        """
215        try:
216            dtype = self.dataCombo.currentText()
217            algo_name = self.algoCombo.currentText()
218            out_fmt = self.outputCombo.currentText()
219
220            data = self.load_data(dtype)
221            nb_cols = self.count_columns(data, dtype)
222
223            prefs = None
224            if algo_name in ("RankSky", "CoskyAlgorithme", "CoskySQL"):
225                prompt_msg = (
226                    f"{nb_cols} columns detected.\n"
227                    "For each column, enter 'min' (lower is better) or 'max' (higher is better),\n"
228                    "separated by commas.\n\n"
229                    "Example: " + ",".join(["min" if i % 2 == 0 else "max" for i in range(min(3, nb_cols))]) + (",..." if nb_cols > 3 else "") + "\n\n"
230                    f"Enter {nb_cols} values:"
231                )
232                s, ok = QInputDialog.getText(self, "Preferences", prompt_msg)
233                if not ok or not s: return
234                parts = [p.strip().lower() for p in s.split(",")]
235                if len(parts) != nb_cols or any(p not in ("min", "max") for p in parts):
236                    QMessageBox.critical(self, "Error", "Incorrect preference list.")
237                    return
238                prefs = [Preference.MIN if p == "min" else Preference.MAX for p in parts]
239
240            algo_cls = {"SkyIR": SkyIR, "DpIdpDh": DpIdpDh,
241                        "CoskyAlgorithme": CoskyAlgorithme, "CoskySQL": CoskySQL,
242                        "RankSky": RankSky}[algo_name]
243
244            exporter = CsvExporterImpl("Assets/Export/CSVFiles/Results.csv") if out_fmt == "CSV" \
245                      else JsonExporterImpl("Assets/Export/JsonFiles/Result.json")
246
247            app = App(data, algo_cls, exporter=exporter,
248                      input_type=dtype, input_file=self.selectedDataPath or "inline",
249                      preferences=prefs)
250            self.lastAppInstance = app
251            self.statusLabel.setText(f"Done in {app.execution_time}s")
252            
253            all_pts = self.get_all_points()
254            sky_pts = self.get_skyline_points()
255            
256            self.display_graph(all_pts, sky_pts)
257
258        except Exception as e:
259            QMessageBox.critical(self, "Error", str(e))
260            self.statusLabel.setText(f"Error: {e}")

262    def _extract_all_points(self, data_obj):
263        """
264        Extracts all data points from a DictObject, JsonObject, or DbObject
265        and returns them as a dictionary mapping each point ID to its list of coordinates.
266        """
267        if isinstance(data_obj, DictObject): 
268            return {k: list(v) for k, v in data_obj.r.items()}
269        if isinstance(data_obj, JsonObject): 
270            return {k: list(v) for k, v in readJson(data_obj.fp).items()}
271        if isinstance(data_obj, DbObject):
272            con = sqlite3.connect(data_obj.fp)
273            cur = con.cursor()
274            cur.execute(f"SELECT * FROM {self.tableName}")
275            rows = {row[0]: list(row[1:]) for row in cur.fetchall()}
276            con.close()
277            return rows
278        return {}

281    def get_all_points(self) -> dict:
282        """
283        Returns all dataset points in unified coordinate space as a dict
284        mapping each point ID to its list of coordinates ``{id: [coords...]}``.  
285        For RankSky, re-applies the original preferences via ``DataUnifier``
286        to ensure consistency with the coordinate space used for skyline points.
287        Returns an empty dict if no algorithm has been run yet.
288        """
289        if not self.lastAppInstance: return {}
290        
291        algo_name = getattr(self.lastAppInstance, 'algo', None)
292        inst = getattr(self.lastAppInstance, 'algo_instance', None)
293        
294        # CoskyAlgorithme stores a pre-unified dataset in inst.r; use it directly
295        # to guarantee coordinate consistency with the skyline points.
296        if algo_name in ("CoskyAlgorithme",) and inst and hasattr(inst, 'r'):
297            return {k: list(v) for k, v in inst.r.items()}
298        
299        if algo_name == "RankSky" and inst:
300            # RankSky mutates inst.pref to [MAX, MAX, ...] after execution.
301            # prefInit and rTupleInit preserve the original values and are used
302            # here to re-apply the "auto" unification mode for correct display.
303            raw = {k: list(v) for k, v in getattr(inst, 'rTupleInit', {}).items()}
304            prefs_orig = list(getattr(inst, 'prefInit', []))
305            if raw and prefs_orig:
306                unifier = DataUnifier(raw, prefs_orig, mode="auto")
307                return unifier.unify()
308            return raw
309
310        # For all other algorithms, raw data is loaded and unified using the stored preferences.
311        data_obj = getattr(self.lastAppInstance, 'r', None)
312        
313        if data_obj is None:
314            dtype = self.dataCombo.currentText()
315            data_wrapper = self.load_data(dtype)
316            all_pts = self._extract_all_points(data_wrapper)
317        else:
318            all_pts = {k: list(v) for k, v in data_obj.items()}
319
320        prefs = getattr(self.lastAppInstance, 'pref', None)
321        
322        if prefs:
323            unifier = DataUnifier(all_pts, list(prefs), mode="auto")
324            all_pts = unifier.unify()
325            
326        return all_pts

328    def get_skyline_points(self):
329        """
330        Returns the skyline points produced by the last algorithm run,
331        normalized to a common format: ``{id: {'coords': [v1, v2, ...], 'score': value}}``.
332        Each algorithm exposes its results through a different attribute;
333        this method handles the extraction for SkyIR, DpIdpDh, CoskyAlgorithme,
334        CoskySQL, and RankSky. Returns an empty dict if no algorithm has been run.
335        """
336        if not self.lastAppInstance: return {}
337        algo = self.lastAppInstance.algo
338        inst = self.lastAppInstance.algo_instance
339        
340        result_data = {}
341        
342        if algo == "SkyIR":
343            # SkyIR stores results as [(id, score), ...] in inst.result
344            for item in getattr(inst, 'result', []):
345                p_id, score = item[0], item[1]
346                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
347                result_data[p_id] = {'coords': coords, 'score': score}
348
349        elif algo == "DpIdpDh":
350            # DpIdpDh stores scores as {id: score} in inst.score
351            for p_id, score in getattr(inst, 'score', {}).items():
352                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
353                result_data[p_id] = {'coords': coords, 'score': score}
354
355        elif algo == "CoskyAlgorithme":
356            # CoskyAlgorithme stores scores as {id: [..., score]} in inst.s
357            s_dict = getattr(inst, "s", {})
358            for p_id in s_dict:
359                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
360                result_data[p_id] = {'coords': coords, 'score': s_dict[p_id][-1]}
361
362        elif algo == "CoskySQL":
363            # CoskySQL stores results as {id: [coords..., score]} in inst.dict
364            for p_id, vals in getattr(inst, 'dict', {}).items():
365                result_data[p_id] = {'coords': vals[:-1], 'score': vals[-1]}
366
367        elif algo == "RankSky":
368            # RankSky internally unifies to MIN space; rTupleInit and prefInit
369            # are used to re-apply "auto" unification and match the coordinate
370            # space of all_pts for consistent visualization.
371            s_dict = getattr(inst, 'score', {})
372            raw = {k: list(v) for k, v in getattr(inst, 'rTupleInit', {}).items()}
373            prefs_list = list(getattr(inst, 'prefInit', []))
374            if raw and prefs_list:
375                unifier = DataUnifier(raw, prefs_list, mode="auto")
376                raw = unifier.unify()
377            for p_id in s_dict:
378                coords = list(raw.get(p_id) or raw.get(str(p_id)) or [])
379                result_data[p_id] = {'coords': coords, 'score': s_dict[p_id][-1]}
380
381        return result_data

412    def display_graph(self, all_pts, sky_pts):
413        """
414        Renders a 3D scatter plot of the unified dataset on the embedded Matplotlib canvas.
415        Skyline points are drawn in red (with a Pareto front surface when at least 4 points
416        are available); all other points are drawn in blue. Only 3D datasets are supported;
417        a message is displayed for other dimensionalities.
418        """
419        if not all_pts:
420            return
421
422        dim = len(next(iter(all_pts.values())))
423        if dim != 3:
424            self.figure.clear()
425            ax = self.figure.add_subplot(111)
426            ax.text(0.5, 0.5, f"Visualization only supported for 3D data\n(Current: {dim}D)",
427                    ha='center', va='center', transform=ax.transAxes)
428            self.canvas.draw()
429            return
430
431        self.figure.clear()
432        ax = self.figure.add_subplot(111, projection='3d')
433
434        # String-key normalization ensures consistent lookups across algorithms
435        all_pts_str = {str(k): v for k, v in all_pts.items()}
436        sky_ids_str = {str(k) for k in sky_pts.keys()}
437        all_ids_str = set(all_pts_str.keys())
438        other_ids_str = all_ids_str - sky_ids_str
439
440        if other_ids_str:
441            other_coords = np.array([all_pts_str[i] for i in other_ids_str])
442            ax.scatter(other_coords[:, 0], other_coords[:, 1], other_coords[:, 2],
443                       c='blue', label='Others', s=20, alpha=0.6)
444
445        if sky_ids_str:
446            sky_coords = np.array([all_pts_str[i] for i in sky_ids_str if i in all_pts_str])
447            if sky_coords.size > 0:
448                ax.scatter(sky_coords[:, 0], sky_coords[:, 1], sky_coords[:, 2],
449                           c='red', label='Skyline', s=50, edgecolors='black')
450
451                # Pareto front surface via Delaunay triangulation; requires at least 4 non-collinear points
452                if len(sky_coords) >= 4:
453                    try:
454                        ax.plot_trisurf(sky_coords[:, 0], sky_coords[:, 1], sky_coords[:, 2],
455                                        color='red', alpha=0.3)
456                    except Exception as e:
457                        print(f"Pareto surface plotting failed (triangulation): {e}")
458
459        # Axis labels and title
460        ax.set_xlabel('Dim 1')
461        ax.set_ylabel('Dim 2')
462        ax.set_zlabel('Dim 3')
463        ax.set_title(f'Skyline Visualization (3D)')
464        ax.legend()
465        
466        self.canvas.draw()

282    def get_all_points(self) -> dict:
283        """
284        Returns all dataset points in unified coordinate space as a dict
285        mapping each point ID to its list of coordinates ``{id: [coords...]}``.  
286        For RankSky, re-applies the original preferences via ``DataUnifier``
287        to ensure consistency with the coordinate space used for skyline points.
288        Returns an empty dict if no algorithm has been run yet.
289        """
290        if not self.lastAppInstance: return {}
291        
292        algo_name = getattr(self.lastAppInstance, 'algo', None)
293        inst = getattr(self.lastAppInstance, 'algo_instance', None)
294        
295        # CoskyAlgorithme stores a pre-unified dataset in inst.r; use it directly
296        # to guarantee coordinate consistency with the skyline points.
297        if algo_name in ("CoskyAlgorithme",) and inst and hasattr(inst, 'r'):
298            return {k: list(v) for k, v in inst.r.items()}
299        
300        if algo_name == "RankSky" and inst:
301            # RankSky mutates inst.pref to [MAX, MAX, ...] after execution.
302            # prefInit and rTupleInit preserve the original values and are used
303            # here to re-apply the "auto" unification mode for correct display.
304            raw = {k: list(v) for k, v in getattr(inst, 'rTupleInit', {}).items()}
305            prefs_orig = list(getattr(inst, 'prefInit', []))
306            if raw and prefs_orig:
307                unifier = DataUnifier(raw, prefs_orig, mode="auto")
308                return unifier.unify()
309            return raw
310
311        # For all other algorithms, raw data is loaded and unified using the stored preferences.
312        data_obj = getattr(self.lastAppInstance, 'r', None)
313        
314        if data_obj is None:
315            dtype = self.dataCombo.currentText()
316            data_wrapper = self.load_data(dtype)
317            all_pts = self._extract_all_points(data_wrapper)
318        else:
319            all_pts = {k: list(v) for k, v in data_obj.items()}
320
321        prefs = getattr(self.lastAppInstance, 'pref', None)
322        
323        if prefs:
324            unifier = DataUnifier(all_pts, list(prefs), mode="auto")
325            all_pts = unifier.unify()
326            
327        return all_pts

329    def get_skyline_points(self):
330        """
331        Returns the skyline points produced by the last algorithm run,
332        normalized to a common format: ``{id: {'coords': [v1, v2, ...], 'score': value}}``.
333        Each algorithm exposes its results through a different attribute;
334        this method handles the extraction for SkyIR, DpIdpDh, CoskyAlgorithme,
335        CoskySQL, and RankSky. Returns an empty dict if no algorithm has been run.
336        """
337        if not self.lastAppInstance: return {}
338        algo = self.lastAppInstance.algo
339        inst = self.lastAppInstance.algo_instance
340        
341        result_data = {}
342        
343        if algo == "SkyIR":
344            # SkyIR stores results as [(id, score), ...] in inst.result
345            for item in getattr(inst, 'result', []):
346                p_id, score = item[0], item[1]
347                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
348                result_data[p_id] = {'coords': coords, 'score': score}
349
350        elif algo == "DpIdpDh":
351            # DpIdpDh stores scores as {id: score} in inst.score
352            for p_id, score in getattr(inst, 'score', {}).items():
353                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
354                result_data[p_id] = {'coords': coords, 'score': score}
355
356        elif algo == "CoskyAlgorithme":
357            # CoskyAlgorithme stores scores as {id: [..., score]} in inst.s
358            s_dict = getattr(inst, "s", {})
359            for p_id in s_dict:
360                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
361                result_data[p_id] = {'coords': coords, 'score': s_dict[p_id][-1]}
362
363        elif algo == "CoskySQL":
364            # CoskySQL stores results as {id: [coords..., score]} in inst.dict
365            for p_id, vals in getattr(inst, 'dict', {}).items():
366                result_data[p_id] = {'coords': vals[:-1], 'score': vals[-1]}
367
368        elif algo == "RankSky":
369            # RankSky internally unifies to MIN space; rTupleInit and prefInit
370            # are used to re-apply "auto" unification and match the coordinate
371            # space of all_pts for consistent visualization.
372            s_dict = getattr(inst, 'score', {})
373            raw = {k: list(v) for k, v in getattr(inst, 'rTupleInit', {}).items()}
374            prefs_list = list(getattr(inst, 'prefInit', []))
375            if raw and prefs_list:
376                unifier = DataUnifier(raw, prefs_list, mode="auto")
377                raw = unifier.unify()
378            for p_id in s_dict:
379                coords = list(raw.get(p_id) or raw.get(str(p_id)) or [])
380                result_data[p_id] = {'coords': coords, 'score': s_dict[p_id][-1]}
381
382        return result_data

413    def display_graph(self, all_pts, sky_pts):
414        """
415        Renders a 3D scatter plot of the unified dataset on the embedded Matplotlib canvas.
416        Skyline points are drawn in red (with a Pareto front surface when at least 4 points
417        are available); all other points are drawn in blue. Only 3D datasets are supported;
418        a message is displayed for other dimensionalities.
419        """
420        if not all_pts:
421            return
422
423        dim = len(next(iter(all_pts.values())))
424        if dim != 3:
425            self.figure.clear()
426            ax = self.figure.add_subplot(111)
427            ax.text(0.5, 0.5, f"Visualization only supported for 3D data\n(Current: {dim}D)",
428                    ha='center', va='center', transform=ax.transAxes)
429            self.canvas.draw()
430            return
431
432        self.figure.clear()
433        ax = self.figure.add_subplot(111, projection='3d')
434
435        # String-key normalization ensures consistent lookups across algorithms
436        all_pts_str = {str(k): v for k, v in all_pts.items()}
437        sky_ids_str = {str(k) for k in sky_pts.keys()}
438        all_ids_str = set(all_pts_str.keys())
439        other_ids_str = all_ids_str - sky_ids_str
440
441        if other_ids_str:
442            other_coords = np.array([all_pts_str[i] for i in other_ids_str])
443            ax.scatter(other_coords[:, 0], other_coords[:, 1], other_coords[:, 2],
444                       c='blue', label='Others', s=20, alpha=0.6)
445
446        if sky_ids_str:
447            sky_coords = np.array([all_pts_str[i] for i in sky_ids_str if i in all_pts_str])
448            if sky_coords.size > 0:
449                ax.scatter(sky_coords[:, 0], sky_coords[:, 1], sky_coords[:, 2],
450                           c='red', label='Skyline', s=50, edgecolors='black')
451
452                # Pareto front surface via Delaunay triangulation; requires at least 4 non-collinear points
453                if len(sky_coords) >= 4:
454                    try:
455                        ax.plot_trisurf(sky_coords[:, 0], sky_coords[:, 1], sky_coords[:, 2],
456                                        color='red', alpha=0.3)
457                    except Exception as e:
458                        print(f"Pareto surface plotting failed (triangulation): {e}")
459
460        # Axis labels and title
461        ax.set_xlabel('Dim 1')
462        ax.set_ylabel('Dim 2')
463        ax.set_zlabel('Dim 3')
464        ax.set_title(f'Skyline Visualization (3D)')
465        ax.legend()
466        
467        self.canvas.draw()

329    def get_skyline_points(self):
330        """
331        Returns the skyline points produced by the last algorithm run,
332        normalized to a common format: ``{id: {'coords': [v1, v2, ...], 'score': value}}``.
333        Each algorithm exposes its results through a different attribute;
334        this method handles the extraction for SkyIR, DpIdpDh, CoskyAlgorithme,
335        CoskySQL, and RankSky. Returns an empty dict if no algorithm has been run.
336        """
337        if not self.lastAppInstance: return {}
338        algo = self.lastAppInstance.algo
339        inst = self.lastAppInstance.algo_instance
340        
341        result_data = {}
342        
343        if algo == "SkyIR":
344            # SkyIR stores results as [(id, score), ...] in inst.result
345            for item in getattr(inst, 'result', []):
346                p_id, score = item[0], item[1]
347                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
348                result_data[p_id] = {'coords': coords, 'score': score}
349
350        elif algo == "DpIdpDh":
351            # DpIdpDh stores scores as {id: score} in inst.score
352            for p_id, score in getattr(inst, 'score', {}).items():
353                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
354                result_data[p_id] = {'coords': coords, 'score': score}
355
356        elif algo == "CoskyAlgorithme":
357            # CoskyAlgorithme stores scores as {id: [..., score]} in inst.s
358            s_dict = getattr(inst, "s", {})
359            for p_id in s_dict:
360                coords = list(inst.r.get(p_id) or inst.r.get(str(p_id)) or [])
361                result_data[p_id] = {'coords': coords, 'score': s_dict[p_id][-1]}
362
363        elif algo == "CoskySQL":
364            # CoskySQL stores results as {id: [coords..., score]} in inst.dict
365            for p_id, vals in getattr(inst, 'dict', {}).items():
366                result_data[p_id] = {'coords': vals[:-1], 'score': vals[-1]}
367
368        elif algo == "RankSky":
369            # RankSky internally unifies to MIN space; rTupleInit and prefInit
370            # are used to re-apply "auto" unification and match the coordinate
371            # space of all_pts for consistent visualization.
372            s_dict = getattr(inst, 'score', {})
373            raw = {k: list(v) for k, v in getattr(inst, 'rTupleInit', {}).items()}
374            prefs_list = list(getattr(inst, 'prefInit', []))
375            if raw and prefs_list:
376                unifier = DataUnifier(raw, prefs_list, mode="auto")
377                raw = unifier.unify()
378            for p_id in s_dict:
379                coords = list(raw.get(p_id) or raw.get(str(p_id)) or [])
380                result_data[p_id] = {'coords': coords, 'score': s_dict[p_id][-1]}
381
382        return result_data
